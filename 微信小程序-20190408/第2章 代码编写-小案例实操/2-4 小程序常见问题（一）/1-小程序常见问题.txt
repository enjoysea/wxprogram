三、发起请求：API → 网络
API请求实例
onLoad: function () {  
    this.getProList()
}

wx.request({
      url: 'http://127.0.0.0:5000',
      method:'GET',
      success: function(res){
        console.log(res)
        self.setData({
          proList: res.data
        })
      }
})
